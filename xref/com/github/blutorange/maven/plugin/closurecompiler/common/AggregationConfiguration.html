14  package com.github.blutorange.maven.plugin.closurecompiler.common;
15  
16  import java.util.List;
17  
18  /** Maps aggregations from an external JSON formatted config file defined in the option {@code bundleConfiguration}. */
19  public class AggregationConfiguration {
20  
21      private List<Aggregation> bundles;
22  
23      /**
24       * Gets the bundles.
25       *
26       * @return the bundles
27       */
28      public List<Aggregation> getBundles() {
29          return bundles;
30      }
31  
32      /**
33       * Sets the bundles.
34       *
35       * @param bundles the bundles to set
36       */
37      public void setBundles(List<Aggregation> bundles) {
38          this.bundles = bundles;
39      }
40  }
