37      private static class EncodingProvider {
38          private final File basedir;
39          private final Map<String, Charset> encodingMap = new HashMap<>();
40  
41          EncodingProvider(File basedir) throws IOException {
42              this.basedir = basedir;
43              var encodingData = new File(basedir, "encoding.txt");
44              if (!encodingData.exists()) {
45                  return;
46              }
47  
48              var encodingLines = FileUtils.readLines(encodingData, UTF_8);
49              for (var encodingLine : encodingLines) {
50                  var parts = encodingLine.split("=");
51                  if (parts.length == 2) {
52                      var relativePath = parts[0];
53                      var encoding = Charset.forName(parts[1]);
54                      encodingMap.put(relativePath, encoding);
55                  }
56              }
57          }
58  
59          public Charset determineEncoding(File file) {
60              var relativePath = relativizePath(basedir, file);
61              return encodingMap.getOrDefault(relativePath, UTF_8);
62          }
63      }

87      private void assertDirContent(File basedir) throws IOException {
88          var actual = new File(new File(basedir, "target"), "test");
89          var expected = new File(basedir, "expected");
90          assertThat(expected.exists())
91                  .as("Expected directory %s must exist", expected)
92                  .isTrue();
93  
94          var expectedFiles = listFiles(expected);
95          assertThat(expectedFiles)
96                  .as("There must be at least one expected file. Add a file 'nofiles' if you expect there to be no files")
97                  .isNotEmpty();
98  
99          if (expectedFiles.size() == 1
100                 && "nofiles".equals(expectedFiles.values().iterator().next().getName())) {
101             if (actual.exists()) {
102                 assertThat(listFiles(actual))
103                         .as("Expecting no output files in %s", actual)
104                         .isEmpty();
105             }
106             return;
107         }
108 
109         assertThat(actual.exists()).as("Actual directory %s must exist", actual).isTrue();
110 
111         var actualFiles = listFiles(actual);
112         assertThat(actualFiles).containsOnlyKeys(expectedFiles.keySet());
113 
114         var encodingProvider = new EncodingProvider(basedir);
115         expectedFiles.forEach((key, expectedFile) -> {
116             var actualFile = actualFiles.get(key);
117             assertThatNoException()
118                     .isThrownBy(() ->
119                             compareFiles(expectedFile, actualFile, encodingProvider.determineEncoding(expectedFile)));
120         });
121     }

123     private void clean(File basedir) throws IOException {
124         var target = new File(basedir, "target");
125         if (target.exists()) {
126             FileUtils.forceDelete(target);
127         }
128         assertThat(target.exists())
129                 .as("Target folder must not exist anymore: %s", target)
130                 .isFalse();
131     }

133     private void compareFiles(File expectedFile, File actualFile, Charset charset) throws IOException {
134         final List<String> expectedLines;
135         final List<String> actualLines;
136 
137         assertThat(expectedFile).exists();
138         assertThat(actualFile).exists();
139 
140         if (expectedFile.getAbsolutePath().endsWith(".gz")) {
141             expectedLines = readLinesFromGzipFile(expectedFile, charset);
142             actualLines = readLinesFromGzipFile(actualFile, charset);
143         } else {
144             expectedLines = FileUtils.readLines(expectedFile, charset);
145             actualLines = FileUtils.readLines(actualFile, charset);
146         }
147 
148         // Ignore empty lines
149         expectedLines.removeIf(StringUtils::isBlank);
150         actualLines.removeIf(StringUtils::isBlank);
151 
152         // Check file contents
153         assertThat(expectedLines).as("Content of %s", expectedFile).isNotEmpty();
154         assertThat(actualLines)
155                 .as("Content of %s has same number of non-empty lines as %s", actualFile, expectedFile)
156                 .hasSameSizeAs(expectedLines);
157 
158         var lines = expectedLines.size();
159         for (int i = 0; i < lines; ++i) {
160             assertThat(actualLines.get(i))
161                     .as("Line from %s should be the same as line from %s", actualFile, expectedFile)
162                     .isEqualToIgnoringNewLines(expectedLines.get(i));
163         }
164     }

166     private List<String> readLinesFromGzipFile(File file, Charset charset) throws IOException {
167         try (var fileInputStream = new FileInputStream(file);
168                 var gzipInputStream = new GZIPInputStream(fileInputStream)) {
169 
170             return IOUtils.readLines(gzipInputStream, charset);
171         }
172     }

174     private MavenResult invokeMaven(File pom, String goal, Collection<String> profiles) throws IOException {
175         final var args = new ArrayList<String>();
176         args.add("clean");
177         args.add(goal);
178         args.add("-DskipTests");
179         profiles.stream().flatMap(profile -> Stream.of("-P", profile)).forEach(args::add);
180         System.setProperty("maven.multiModuleProjectDirectory", pom.getParent());
181         LOG.info("Invoking maven: {}", StringUtils.join(args, " "));
182         try (final var out = new ByteArrayOutputStream()) {
183             // We don't want to close ChainedPrintStream as it would close System.out, so we don't use
184             // try-with-resources
185             final var outStream = new ChainedPrintStream(new PrintStream(out), System.out);
186             new MavenCli().doMain(args.toArray(new String[0]), pom.getParent(), outStream, outStream);
187             return new MavenResult(out.toString(UTF_8));
188         }
189     }

191     private Map<String, File> listFiles(File basedir) {
192         return FileUtils.listFiles(basedir, null, true).stream()
193                 .collect(Collectors.toMap(file -> relativizePath(basedir, file), identity()));
194     }

196     private MavenResult runMinify(String projectName, Collection<String> profiles) throws Exception {
197         final var parentDir = testResources.getBasedir("parent").getCanonicalFile();
198         final var parentPom = new File(parentDir, "pom.xml");
199         final var parentPomNew = new File(parentDir.getParentFile(), "pom.xml");
200         assertThat(parentPom).exists();
201         FileUtils.copyFile(parentPom, parentPomNew);
202 
203         final var basedir = testResources.getBasedir(projectName).getCanonicalFile();
204         final var pom = new File(basedir, "pom.xml");
205         assertThat(pom).exists();
206 
207         clean(basedir);
208         invokeMaven(parentPomNew, "install", List.of());
209         return invokeMaven(pom, "package", profiles);
210     }

343     public void testSkipIfExists() throws Exception {
344         runMinify("skipif", List.of("skipIfExists"));
345 
346         // This creates the (older) output file, so the minification process should not run
347         runMinifyAndAssertDirContent("skipif", profiles("createOlderFile", "skipIfExists"));
348 
349         // Now force is enabled, minification should run
350         assertThatThrownBy(() ->
351                         runMinifyAndAssertDirContent("skipif", profiles("createOlderFile", "skipIfExists", "force")))
352                 .isInstanceOf(AssertionError.class);
