44      private static class EncodingProvider {
45          private final File basedir;
46          private final Map<String, Charset> encodingMap = new HashMap<>();
47  
48          EncodingProvider(File basedir) throws IOException {
49              this.basedir = basedir;
50              var encodingData = new File(basedir, "encoding.txt");
51              if (encodingData.exists()) {
52                  var encodingLines = FileUtils.readLines(encodingData, UTF_8);
53                  for (var encodingLine : encodingLines) {
54                      var parts = encodingLine.split("=");
55                      if (parts.length == 2) {
56                          var relativePath = parts[0];
57                          var encoding = Charset.forName(parts[1]);
58                          encodingMap.put(relativePath, encoding);
59                      }
60                  }
61              }
62          }
63  
64          public Charset determineEncoding(File file) {
65              var relativePath = relativizePath(basedir, file);
66              return encodingMap.getOrDefault(relativePath, UTF_8);
67          }
68      }

93      private void assertDirContent(File basedir) throws IOException {
94          var expected = new File(basedir, "expected");
95          var actual = new File(new File(basedir, "target"), "test");
96          var expectedFiles = expected.exists() ? listFiles(expected) : new HashMap<String, File>();
97          var actualFiles = actual.exists() ? listFiles(actual) : new HashMap<String, File>();
98          LOG.info("Comparing actual files [\n"
99                  + actualFiles.values().stream().map(File::getAbsolutePath).collect(Collectors.joining(",\n")) + "\n]");
100         LOG.info("to the expected files [\n"
101                 + expectedFiles.values().stream().map(File::getAbsolutePath).collect(Collectors.joining(",\n"))
102                 + "\n]");
103         assertFalse(
104                 expectedFiles.isEmpty(),
105                 "There must be at least one expected file. Add a file 'nofiles' if you expect there to be no files");
106         var encodingProvider = new EncodingProvider(basedir);
107         if (expectedFiles.size() == 1
108                 && "nofiles".equals(expectedFiles.values().iterator().next().getName())) {
109             // Expect there to be no output files
110             assertEquals(0, actualFiles.size());
111         } else {
112             assertEquals(
113                     expectedFiles.size(),
114                     actualFiles.size(),
115                     "Number of expected files must match the number of produced files. "
116                             + diffFiles(basedir, expectedFiles, actualFiles));
117             assertTrue(
118                     CollectionUtils.isEqualCollection(expectedFiles.keySet(), actualFiles.keySet()),
119                     "Expected file names must match the produced file names. "
120                             + diffFiles(basedir, expectedFiles, actualFiles));
121             expectedFiles.forEach((key, expectedFile) -> {
122                 var actualFile = actualFiles.get(key);
123                 try {
124                     compareFiles(expectedFile, actualFile, encodingProvider);
125                 } catch (IOException e) {
126                     throw new RuntimeException(e);
127                 }
128             });
129         }
130     }

132     private static String diffFiles(File basedir, Map<String, File> expectedFiles, Map<String, File> actualFiles) {
133         final var testDir = new File(basedir, "target/test");
134         final var expectedButNotPresent = SetUtils.difference(expectedFiles.keySet(), actualFiles.keySet()).stream()
135                 .map(x -> new File(testDir, x))
136                 .map(FileHelper::absoluteFileToCanonicalFile)
137                 .collect(toList());
138         final var presentButNotExpected = SetUtils.difference(actualFiles.keySet(), expectedFiles.keySet()).stream()
139                 .map(actualFiles::get)
140                 .collect(toList());
141         final var messages = new ArrayList<String>();
142         if (!expectedButNotPresent.isEmpty()) {
143             messages.add("Expected files that are not present: <" + expectedButNotPresent + ">.");
144         }
145         if (!presentButNotExpected.isEmpty()) {
146             messages.add("Present files that were not expected to be present: <" + presentButNotExpected + ">");
147         }
148         return String.join(" ", messages);
149     }

151     private void clean(File basedir) throws IOException {

155         }
156         assertFalse(target.exists());

159     private void compareFiles(File expectedFile, File actualFile, EncodingProvider encodingProvider)
160             throws IOException {
161         var encoding = encodingProvider.determineEncoding(expectedFile);
162         var expectedLines = FileUtils.readLines(expectedFile, encoding);
163         var actualLines = FileUtils.readLines(actualFile, encoding);
164         assertTrue(
165                 expectedFile.exists(),
166                 "File with expected content does not exist: '" + actualFile.getAbsolutePath() + "'");
167         assertTrue(
168                 actualFile.exists(),
169                 "File with produced content does not exist: '" + actualFile.getAbsolutePath() + "'");
170         // Ignore empty lines
171         expectedLines.removeIf(StringUtils::isBlank);
172         actualLines.removeIf(StringUtils::isBlank);
173         // Check file contents
174         assertFalse(
175                 expectedLines.isEmpty(),
176                 "Expected file must contain at least one non-empty line: '" + actualFile.getAbsolutePath() + "'");
177         assertEquals(
178                 expectedLines.size(),
179                 actualLines.size(),
180                 "Number of non-empty lines in expected file must match the generated number of lines: '"
181                         + actualFile.getAbsolutePath() + "'");
182         for (int i = 0, j = expectedLines.size(); i < j; ++i) {
183             assertEquals(
184                     expectedLines.get(i).trim(),
185                     actualLines.get(i).trim(),
186                     "Actual content of file '" + actualFile.getAbsolutePath() + "' differs from the expected content");
187         }
188     }

190     private <T extends Throwable> void expectError(Class<T> error, Action runnable) {
191         try {
192             runnable.run();
193         } catch (Throwable e) {
194             if (error.isInstance(e)) {
195                 return;
196             }
197             fail("Action threw an error of type " + e.getClass().getSimpleName()
198                     + ", but it is not of the expected type " + error.getSimpleName());
199         }
200         fail("Action did not throw the expected error type " + error.getSimpleName());
201     }

203     private MavenResult invokeMaven(File pom, String goal, Collection<String> profiles) throws IOException {
204         final var args = new ArrayList<String>();
205         args.add("clean");
206         args.add(goal);
207         args.add("-DskipTests");
208         profiles.stream().flatMap(profile -> Stream.of("-P", profile)).forEach(args::add);
209         System.setProperty("maven.multiModuleProjectDirectory", pom.getParent());
210         LOG.info("Invoking maven: " + StringUtils.join(args, " "));
211         try (final var out = new ByteArrayOutputStream();
212                 final var err = new ByteArrayOutputStream()) {
213             try (final var outStream = new ChainedPrintStream(new PrintStream(out), System.out);
214                     final var errStream = new ChainedPrintStream(new PrintStream(err), System.err)) {
215                 final var cli = new MavenCli();
216                 cli.doMain(args.toArray(new String[0]), pom.getParent(), outStream, errStream);
217                 return new MavenResult(out.toString(UTF_8), err.toString(UTF_8));
218             }
219         }
220     }

223         return FileUtils.listFiles(basedir, null, true).stream()

231     private MavenResult runMinify(String projectName, Collection<String> profiles) throws Exception {
232         final var parentDir = testResources.getBasedir("parent").getCanonicalFile();
233         final var parentPom = new File(parentDir, "pom.xml");
234         final var parentPomNew = new File(parentDir.getParentFile(), "pom.xml");
235         assertTrue(parentPom.exists());
236         FileUtils.copyFile(parentPom, parentPomNew);
237 
238         final var basedir = testResources.getBasedir(projectName).getCanonicalFile();
239         final var pom = new File(basedir, "pom.xml");
240         assertTrue(pom.exists());
241 
242         clean(basedir);
243         invokeMaven(parentPomNew, "install", Collections.emptySet());
244         return invokeMaven(pom, "package", profiles);
245     }

373     @Test
374     public void testSkipIfExists() throws Exception {
375         // Output file does not exist, minification should run
376         expectError(AssertionError.class, () -> runMinifyAndAssertDirContent("skipif", List.of("skipIfExists")));
377 
378         // This creates the (older) output file, so the minification process should not run
379         runMinifyAndAssertDirContent("skipif", List.of("createOlderFile", "skipIfExists"));
380 
381         // Now force is enabled, minification should run
382         expectError(
383                 AssertionError.class,
384                 () -> runMinifyAndAssertDirContent(
385                         "skipif", Arrays.asList("createOlderFile", "skipIfExists", "force")));
386     }
