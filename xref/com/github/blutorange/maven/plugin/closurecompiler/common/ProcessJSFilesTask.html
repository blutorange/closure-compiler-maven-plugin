40  /** Task for merging and compressing JavaScript files. */
41  public final class ProcessJSFilesTask extends ProcessFilesTask {
42  
43      /**
44       * Task constructor.
45       *
46       * @param mojoMeta Mojo meta (for log, project etc.)
47       * @param processConfig Details about the process files task.
48       * @param fileSpecifier Details about the input / output files.
49       * @param closureConfig Google Closure Compiler configuration
50       * @throws IOException When an input file could not be read of an output file could not be written.
51       */
52      public ProcessJSFilesTask(
53              MojoMetadata mojoMeta,
54              FileProcessConfig processConfig,
55              FileSpecifier fileSpecifier,
56              ClosureConfig closureConfig)
57              throws IOException {
58          super(mojoMeta, processConfig, fileSpecifier, closureConfig);
59      }
60  
61      /**
62       * Minifies a JavaScript file. Create missing parent directories if needed.
63       *
64       * @param mergedFile input file resulting from the merged step
65       * @param minifiedFile output file resulting from the minify step
66       * @throws IOException when the minify step fails
67       * @throws MojoFailureException When the file could not be minified, either due to an I/O error or due to a closure
68       *     compiler failure.
69       */
70      @Override
71      protected ProcessingResult minify(File mergedFile, File minifiedFile) throws IOException, MojoFailureException {
72          final var srcFiles = new ArrayList<File>();
73          srcFiles.add(mergedFile);
74          return minify(srcFiles, minifiedFile);
75      }
76  
77      @Override
78      protected ProcessingResult minify(List<File> srcFiles, File minifiedFile) throws IOException, MojoFailureException {
79          final var sourceMapFile = closureConfig
80                  .getSourceMapInterpolator()
81                  .interpolate(minifiedFile, minifiedFile.getParentFile(), minifiedFile.getParentFile());
82  
83          if (!haveFilesChanged(
84                  srcFiles,
85                  closureConfig.isCreateSourceMapFile()
86                          ? Arrays.asList(minifiedFile, sourceMapFile)
87                          : Collections.singleton(minifiedFile))) {
88              return ProcessingResult.skipped(minifiedFile).build();
89          }
90  
91          mkDir(targetDir);
92          mkDir(minifiedFile.getParentFile());
93  
94          if (closureConfig.isCreateSourceMapFile()) {
95              mkDir(sourceMapFile.getParentFile());
96          }
97  
98          final var outputInterpolator = closureConfig.getOutputInterpolator();
99  
100         mojoMeta.getLog().info("Creating the minified file [" + minifiedFile.getName() + "].");
101         mojoMeta.getLog().debug("Full path is [" + minifiedFile.getPath() + "].");
102 
103         final var baseDirForSourceFiles = getBaseDirForSourceFiles(minifiedFile, sourceMapFile);
104 
105         mojoMeta.getLog()
106                 .debug("Setting base dir for closure source files to [" + baseDirForSourceFiles.getAbsolutePath()
107                         + "]");
108 
109         final var sourceFileList = new ArrayList<SourceFile>();
110         for (final var srcFile : srcFiles) {
111             try (InputStream in = new FileInputStream(srcFile)) {
112                 SourceFile input = SourceFile.builder()
113                         .withPath(FileHelper.relativizePath(baseDirForSourceFiles, srcFile))
114                         .withCharset(mojoMeta.getEncoding())
115                         .withContent(in)
116                         .build();
117                 sourceFileList.add(input);
118             }
119         }
120 
121         // Create compiler options
122         final var fileSystemMapping =
123                 new FileSystemLocationMapping(mojoMeta.getLog(), baseDirForSourceFiles, sourceMapFile);
124         final var options = closureConfig.getCompilerOptions(
125                 fileSystemMapping, minifiedFile, sourceMapFile, baseDirForSourceFiles, sourceDir);
126 
127         if (mojoMeta.getLog().isDebugEnabled()) {
128             mojoMeta.getLog()
129                     .debug("Transpiling with closure source files: ["
130                             + sourceFileList.stream().map(SourceFile::toString).collect(Collectors.joining(", "))
131                             + "]");
132             mojoMeta.getLog()
133                     .debug("Transpiling from [" + options.getLanguageIn() + "] to [" + closureConfig.getLanguageOut()
134                             + "], strict=" + options.shouldEmitUseStrict());
135             mojoMeta.getLog().debug("Starting compilations with closure compiler options: " + options);
136         }
137 
138         // Set (external) libraries to be available
139         final var externs = new ArrayList<SourceFile>();
140         externs.addAll(CommandLineRunner.getBuiltinExterns(closureConfig.getEnvironment()));
141         externs.addAll(closureConfig.getExterns());
142 
143         // Now compile
144         final var compiler = new Compiler();
145         compiler.compile(externs, sourceFileList, options);
146 
147         // Check for errors.
148         checkForErrors(compiler, baseDirForSourceFiles);
149 
150         // Write compiled file to output file
151         final var compiled = compiler.toSource();
152 
153         OutputStream output;
154         Writer outputWriter = null;
155         try {
156             output = mojoMeta.getBuildContext().newFileOutputStream(minifiedFile);
157             try {
158                 outputWriter = new OutputStreamWriter(output, mojoMeta.getEncoding());
159             } finally {
160                 // When new OutputStreamWriter threw an exception, writer is null
161                 if (outputWriter == null && output != null) {
162                     output.close();
163                 }
164             }
165             outputWriter.append(outputInterpolator.apply(compiled));
166 
167             // Create source map if configured.
168             if (closureConfig.isCreateSourceMap()) {
169                 // Adjust source map for output wrapper.
170                 compiler.getSourceMap().setWrapperPrefix(outputInterpolator.getWrapperPrefix());
171                 fileSystemMapping.setTranspilationDone(true);
172                 createSourceMap(outputWriter, compiler, minifiedFile, sourceMapFile);
173             }
174 
175             // Make sure we end with a new line
176             outputWriter.append(processConfig.getLineSeparator());
177         } finally {
178             // Closing the OutputStream from m2e as well causes a StreamClosed exception in m2e
179             // So we cannot use a try-with-resource
180             if (outputWriter != null) {
181                 outputWriter.close();
182             }
183         }
184 
185         mojoMeta.getBuildContext().refresh(minifiedFile);
186 
187         logCompressionGains(srcFiles, compiled);
188 
189         return ProcessingResult.success(minifiedFile).build();
190     }
191 
192     private File getBaseDirForSourceFiles(File minifiedFile, File sourceMapFile) {
193         return this.sourceDir;
194     }
195 
196     private void checkForErrors(Compiler compiler, File baseDirForSourceFiles) {
197         // Add warning to build context, so it shows up in IDEs etc.
198         for (final var warning : compiler.getWarnings()) {
199             ClosureCompileFileMessage.ofWarning(warning, compiler, baseDirForSourceFiles)
200                     .addTo(mojoMeta.getBuildContext());
201         }
202 
203         final var errors = new ArrayList<>(compiler.getErrors());
204         if (!errors.isEmpty()) {
205             final var fileErrors = errors.stream()
206                     .map(error -> ClosureCompileFileMessage.ofError(error, compiler, baseDirForSourceFiles));
207             throw new FileException(fileErrors::iterator);
208         }
209     }

211     private void createSourceMap(Writer writer, Compiler compiler, File minifiedFile, File sourceMapFile)
212             throws IOException {
213         final var pathToSource =
214                 FilenameUtils.separatorsToUnix(FileHelper.relativizePath(sourceMapFile.getParentFile(), minifiedFile));
215         mojoMeta.getLog().debug("Setting path to source in source map to [" + pathToSource + "].");
216         switch (closureConfig.getSourceMapOutputType()) {
217             case inline: {
218                 mojoMeta.getLog().info("Creating the inline source map.");
219                 final var sb = new StringBuilder();
220                 compiler.getSourceMap().appendTo(sb, pathToSource);
221                 final var dataUrl = new DataUrlBuilder()
222                         .setMimeType("application/json")
223                         .setEncoding(DataUrlEncoding.BASE64)
224                         .setData(sb.toString().getBytes(StandardCharsets.UTF_8))
225                         .setHeader("charset", "utf-8")
226                         .build();
227                 final var serializer = new DataUrlSerializer();
228                 final var serializedDataUrl = serializer.serialize(dataUrl);
229                 writer.append(processConfig.getLineSeparator());
230                 writer.append("//# sourceMappingURL=").append(serializedDataUrl);
231                 break;
232             }
233             case file:
234                 flushSourceMap(sourceMapFile, pathToSource, compiler.getSourceMap());
235                 break;
236             case reference: {
237                 mojoMeta.getLog().info("Creating reference to source map.");
238                 final var pathToMap = FilenameUtils.separatorsToUnix(
239                         FileHelper.relativizePath(minifiedFile.getParentFile(), sourceMapFile));
240                 flushSourceMap(sourceMapFile, pathToSource, compiler.getSourceMap());
241                 writer.append(processConfig.getLineSeparator());
242                 writer.append("//# sourceMappingURL=").append(pathToMap);
243                 break;
244             }
245             default:
246                 mojoMeta.getLog()
247                         .warn("Unknown source map inclusion type [" + closureConfig.getSourceMapOutputType() + "]");
248                 throw new RuntimeException(
249                         "unknown source map inclusion type: " + closureConfig.getSourceMapOutputType());
250         }
251     }
252 
253     private void flushSourceMap(File sourceMapFile, String pathToSource, SourceMap sourceMap) throws IOException {
254         mojoMeta.getLog().info("Creating the source map [" + sourceMapFile.getName() + "].");
255         mojoMeta.getLog().debug("Full path is [" + sourceMapFile.getPath() + "].");
256 
257         OutputStream output;
258         Writer outputWriter = null;
259         try {
260             output = mojoMeta.getBuildContext().newFileOutputStream(sourceMapFile);
261             try {
262                 outputWriter = new OutputStreamWriter(output, mojoMeta.getEncoding());
263             } finally {
264                 // When new OutputStreamWriter threw an exception, writer is null
265                 if (outputWriter == null && output != null) output.close();
266             }
267 
268             sourceMap.appendTo(outputWriter, pathToSource);
269         } catch (IOException e) {
270             mojoMeta.getLog()
271                     .error("Failed to write the JavaScript Source Map file [" + sourceMapFile.getName() + "].", e);
272             mojoMeta.getLog().debug("Full path is [" + sourceMapFile.getPath() + "]");
273         } finally {
274             // Closing the OutputStream from m2e as well causes a StreamClosed exception in m2e
275             // So we cannot use a try-with-resource
276             if (outputWriter != null) {
277                 outputWriter.close();
278             }
279         }
280     }
