3   import static org.owasp.encoder.Encode.forHtmlAttribute;
4   import static org.owasp.encoder.Encode.forXmlAttribute;

6   import java.util.Objects;
7   import org.jsoup.nodes.Element;
8   
9   final class HtmlModifier {
10      public static TextFileModification clearTextContent(Element element) {
11          final var selfClosed = element.sourceRange().equals(element.endSourceRange());
12          if (selfClosed || element.childNodeSize() == 0) {
13              return null;
14          }
15          final var firstChildSourceRange = element.childNode(0).sourceRange();
16          final var lastChild = element.childNode(element.childNodeSize() - 1);
17          final var lastChildSourceRange =
18                  lastChild instanceof Element ? ((Element) lastChild).endSourceRange() : lastChild.sourceRange();
19          final var from = firstChildSourceRange.startPos();
20          final var to = lastChildSourceRange.endPos();
21          return to > from ? new TextFileModification(from, to, "") : null;
22      }
23  
24      public static TextFileModification setAttribute(Element element, String name, String newValue, boolean html) {
25          final var escapedValue = html ? forHtmlAttribute(newValue) : forXmlAttribute(newValue);
26          final var attributes = element.attributes();
27          final var attributesIterator = attributes.iterator();
28          final var selfClosed = element.sourceRange().equals(element.endSourceRange());
29          while (attributesIterator.hasNext()) {
30              final var attribute = attributesIterator.next();
31              if (Objects.equals(name, attribute.getKey())) {
32                  if (Objects.equals(newValue, attribute.getValue())) {
33                      return null;
34                  }
35                  final var sourceRange = attribute.sourceRange();
36                  final var nameRange = sourceRange.valueRange();
37                  final var valueRange = sourceRange.valueRange();
38                  final var hasValue = element.attributes().hasDeclaredValueForKey(name);
39                  if (hasValue) {
40                      final var badEmptyRange = nameRange.endPos() == valueRange.startPos();
41                      if (badEmptyRange) {
42                          // E.g. for <script src   =    ''>, JSoup does not report the actual position
43                          // of the empty string (''), but the position immediately after the attribute name.
44                          final var nextAttribute = attributesIterator.hasNext() ? attributesIterator.next() : null;
45                          final var endPos = nextAttribute != null
46                                  ? nextAttribute.sourceRange().nameRange().startPos()
47                                  : element.sourceRange().endPos() - (selfClosed ? 2 : 1);
48                          final var quotedValue = "=\"" + escapedValue + "\"" + (nextAttribute != null ? " " : "");
49                          return new TextFileModification(valueRange.startPos(), endPos, quotedValue);
50                      } else {
51                          return new TextFileModification(valueRange.startPos(), valueRange.endPos(), escapedValue);
52                      }
53                  } else {
54                      final var quotedValue = "=\"" + escapedValue + "\"";
55                      return new TextFileModification(valueRange.endPos(), valueRange.endPos(), quotedValue);
56                  }
57              }
58          }
59          // Attribute not found
60          final var endPos = selfClosed
61                  ? element.sourceRange().endPos() - 1
62                  : element.sourceRange().endPos();
63          final var attributeValue = " " + name + "=\"" + escapedValue + "\"";
64          return new TextFileModification(endPos - 1, endPos - 1, attributeValue);
65      }
